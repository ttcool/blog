统计单词表中不包含字母e的单词所占百分比

```python
# -*- coding: utf8 -*-
def has_no_e(file):
    fin = open(file)
    all = []
    words = []
    for line in fin:
        word = line.strip()
        all.append(word)
        if 'e' not in word:
            words.append(word)
    per = float(len(words))/float(len(all))
    print "不包含'e'单词总数:%s" %len(words)
    print "单词总数:%s" %len(all)
    print "百分比:%.2f" %per
has_no_e('words.txt')
```
